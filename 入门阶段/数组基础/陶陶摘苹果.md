## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	int a[10];
	int h, i;
	int s = 0;

	for (i = 0; i < 10; i++)
		cin >> a[i];

	cin >> h;
	h += 30;

	for (i = 0; i < 10; i++) {
		if (h >= a[i])
			s++;
	}

	cout << s;
	return 0;
}
~~~
