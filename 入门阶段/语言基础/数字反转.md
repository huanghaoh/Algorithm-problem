## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	string a;
	int i;
	cin >> a;

	if (a[0] == '-') {
		cout << '-';
		reverse(a.begin() + 1, a.end());
		i = 1;

		while (a[i] == '0') {
			i++;
		}

		for (; i < a.length(); i++)
			cout << a[i];
	} else if (a[0] == '0') {
		cout << '0';
	} else {
		reverse(a.begin(), a.end());
		i = 0;

		while (a[i] == '0') {
			i++;
		}

		for (; i < a.length(); i++)
			cout << a[i];
	}

	return 0;
}
~~~

## 解析

注意反转函数reverse()的使用：

1. 反转字符串：reverse(str.begin(), str.end());
2. 反转数组：reverse(a, a+n);
