## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, a, b, p, t = 3, count;
	cin >> n;

	while (t--) {
		cin >> a >> b;

		if (n % a != 0)
			p = (n / a + 1) * b;
		else
			p = n / a * b;

		if (p < count || t == 2)
			count = p;
	}

	cout << count;
	return 0;
}
~~~

## 解析

注意考虑需不需要多买一份的条件。
