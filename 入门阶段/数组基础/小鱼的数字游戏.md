## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	double a[100] = {0};
	int i;

	for (i = 0; i < 100; i++) {
		cin >> a[i];

		if (a[i] == 0)
			break;
	}

	for (; i >= 0; i--)
		if (a[i] != 0)
			cout << a[i] << " ";

	return 0;
}
~~~
