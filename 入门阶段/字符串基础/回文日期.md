## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	int a, b, i, j, count = 0;
	int year, sum;
	cin >> a >> b;

	for (i = 1; i <= 12; i++) {
		switch (i) {
			case 2:
				for (j = 1; j <= 29; j++) {
					year = i / 10 + i % 10 * 10 + j / 10 * 100 + j % 10 * 1000;
					sum = year * 10000 + i * 100 + j;

					if (sum >= a && sum <= b)
						count++;
				}

				break;

			case 4:
			case 6:
			case 9:
			case 11:
				for (j = 1; j <= 30; j++) {
					year = i / 10 + i % 10 * 10 + j / 10 * 100 + j % 10 * 1000;
					sum = year * 10000 + i * 100 + j;

					if (sum >= a && sum <= b)
						count++;
				}

				break;

			default:
				for (j = 1; j <= 31; j++) {
					year = i / 10 + i % 10 * 10 + j / 10 * 100 + j % 10 * 1000;
					sum = year * 10000 + i * 100 + j;

					if (sum >= a && sum <= b)
						count++;
				}

				break;
		}
	}

	cout << count;
	return 0;
}
~~~

## 解析

本题有三种思路：

1. 遍历输入的日期是不是回文数。（最差）
2. 枚举年份，求月份。
3. 枚举月份，求年份。（最好，不需要讨论是否是闰年）
