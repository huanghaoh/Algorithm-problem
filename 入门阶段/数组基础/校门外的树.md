## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	int l, m, i, s = 0;
	cin >> l >> m;
	int a[l + 1] = {0};
	int u, v;

	while (m--) {
		cin >> u >> v;

		for (i = u; i <= v; i++)
			a[i]++;
	}

	for (i = 0; i <= l; i++) {
		if (a[i] == 0)
			s++;
	}

	cout << s;
	return 0;
}
~~~
