## 答案

~~~c++
#include <bits/stdc++.h>
using namespace std;

int main() {
	string s1[10001], s2[10001];
	int i = 0, j = 0, k, t, count = 0;
	getline(cin, s1[i]);

	while (s1[i] != "EOF") {
		for (k = 0; k < s1[i].length(); k++) {
			if (s1[i][k] == '<') {
				if (k != 0) {
					s1[i].erase(k - 1, 2);
					k -= 2;
				} else {
					s1[i].erase(k, 1);
					k--;
				}
			}
		}

		++i;
		getline(cin, s1[i]);
	}

	getline(cin, s2[j]);

	while (s2[j] != "EOF") {
		for (k = 0; k < s2[j].length(); k++) {
			if (s2[j][k] == '<') {
				if (k != 0) {
					s2[j].erase(k - 1, 2);
					k -= 2;
				} else {
					s2[j].erase(k, 1);
					k--;
				}
			}
		}

		++j;
		getline(cin, s2[j]);
	}

	if (i > j) {//保证行号一致
		for (k = j; k < i; k++)
			s1[k] = "";
	} else if (i < j) {
		for (k = i; k = j; k++)
			s2[k] = "";
	}

	cin >> t;

	for (j = 0; j < i; j++) {
		int m = min(s1[j].length(), s2[j].length());

		for (k = 0; k < m; k++) {
			if (s1[j][k] == s2[j][k])
				count++;
		}
	}

	cout << int(count * 60.0 / t + 0.5);
	return 0;
}
~~~

## 解析

str.erase()可以删除一个字符串中的指定字符。

注意保证范文和输入的行号一致，多余的需要去掉。
